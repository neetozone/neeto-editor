# Formik with neetoEditor — v2

Use the `FormikEditor` component to integrate neetoEditor with formik. The v2
build wires the editor to a `Label` from `@bigbinary/neeto-atoms` for the
form-field label.

### Usage

```jsx
import React from "react";

import { isEditorEmpty, FormikEditor } from "@bigbinary/neeto-editor/v2";
import { Formik, Form } from "formik";
import * as yup from "yup";

const INITIAL_VALUES = {
  description: "<p></p>",
};

const VALIDATION_SCHEMA = yup.object().shape({
  description: yup
    .string()
    .test(
      "description",
      "Description is required",
      value => !isEditorEmpty(value)
    ),
});

const App = () => (
  <Formik
    initialValues={INITIAL_VALUES}
    validationSchema={VALIDATION_SCHEMA}
    onSubmit={values => console.log(values)}
  >
    <Form>
      <FormikEditor required label="Description" name="description" />
      <button type="submit">Submit</button>
    </Form>
  </Formik>
);
```
